\section{–ù–∞–≤—á–∞–Ω–Ω—è –∑ –ø—ñ–¥—Å–∏–ª–µ–Ω–Ω—è–º (reinforcement learning)}

\subsection{–ú–æ–¥–µ–ª—å –Ω–∞–≤—á–∞–Ω–Ω—è –∑ –ø—ñ–¥—Å–∏–ª–µ–Ω–Ω—è–º}

\emph{–ê–≥–µ–Ω—Ç (agent)} —Ç–∞ \emph{—Å–µ—Ä–µ–¥–æ–≤–∏—â–µ (environment)} –≤–∑–∞—î–º–æ–¥—ñ—é—Ç—å –≤ –∫–æ–∂–µ–Ω –¥–∏—Å–∫—Ä–µ—Ç–Ω–∏–π –º–æ–º–µ–Ω—Ç —á–∞—Å—É $t=0,1,2,\dots$. –ù–∞ –∫–æ–∂–Ω–æ–º—É –∫—Ä–æ—Ü—ñ $t$ –∞–≥–µ–Ω—Ç –æ—Ç—Ä–∏–º—É—î –ø–µ–≤–Ω–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è \emph{—Å—Ç–∞–Ω—É —Å–∏—Å—Ç–µ–º–∏ (state)} $s_t \in \mathcal{S}$, –¥–µ $\mathcal{S}$ "--- –º–Ω–æ–∂–∏–Ω–∞ —É—Å—ñ—Ö —Å—Ç–∞–Ω—ñ–≤ —Å–∏—Å—Ç–µ–º–∏, —ñ –Ω–∞ –±–∞–∑—ñ —Ü—å–æ–≥–æ –≤–∏–±–∏—Ä–∞—î \emph{–¥—ñ—é (action)} $a_t \in \mathcal{A}(s_t)$, –¥–µ $\mathcal{A}(s_t)$ "--- –º–Ω–æ–∂–∏–Ω–∞ –¥–æ–ø—É—Å—Ç–∏–º–∏—Ö –¥—ñ–π —É —Å—Ç–∞–Ω—ñ $s_t$. –ù–∞ –Ω–∞—Å—Ç—É–ø–Ω–æ–º—É –∫—Ä–æ—Ü—ñ $t+1$, —á–∞—Å—Ç–∫–æ–≤–æ —è–∫ –Ω–∞—Å–ª—ñ–¥–æ–∫ –≤–∏–∫–æ–Ω–∞–Ω–æ—ó –¥—ñ—ó, –∞–≥–µ–Ω—Ç –æ—Ç—Ä–∏–º—É—î —á–∏—Å–ª–æ–≤—É \emph{–≤–∏–Ω–∞–≥–æ—Ä–æ–¥—É (reward)} $r_{t+1} \in \mathbb{R}$ —ñ –æ–ø–∏–Ω—è—î—Ç—å—Å—è –≤ —Å—Ç–∞–Ω—ñ $s_{t+1}$.

–ù–∞ –∫–æ–∂–Ω–æ–º—É –∫—Ä–æ—Ü—ñ –∞–≥–µ–Ω—Ç —Ä–µ–∞–ª—ñ–∑–æ–≤—É—î –≤—ñ–¥–æ–±—Ä–∞–∂–µ–Ω–Ω—è —Å—Ç–∞–Ω—ñ–≤ —Å–∏—Å—Ç–µ–º–∏ –Ω–∞ –π–º–æ–≤—ñ—Ä–Ω–æ—Å—Ç—ñ –∑–¥—ñ–π—Å–Ω–µ–Ω–Ω—è –∫–æ–∂–Ω–æ—ó –º–æ–∂–ª–∏–≤–æ—ó –¥—ñ—ó. –¢–∞–∫–µ –≤—ñ–¥–æ–±—Ä–∞–∂–µ–Ω–Ω—è –Ω–∞–∑–∏–≤–∞—î—Ç—å—Å—è \emph{—Å—Ç—Ä–∞—Ç–µ–≥—ñ—î—é (policy)} –∞–≥–µ–Ω—Ç–∞ —ñ –ø–æ–∑–Ω–∞—á–∞—î—Ç—å—Å—è $\pi_t$, –¥–µ $\pi(s,a)$ "--- —Ü–µ –π–º–æ–≤—ñ—Ä–Ω—ñ—Å—Ç—å, —â–æ $a_t=a$, —è–∫—â–æ $s_t = s$. –ú–µ—Ç–æ–¥–∏ –Ω–∞–≤—á–∞–Ω–Ω—è –∑ –ø—ñ–¥—Å–∏–ª–µ–Ω–Ω—è–º –≤–∏–∑–Ω–∞—á–∞—é—Ç—å, —è–∫ —Å–∞–º–µ –∞–≥–µ–Ω—Ç –∑–º—ñ–Ω—é—î —Å–≤–æ—é —Å—Ç—Ä–∞—Ç–µ–≥—ñ—é –≤ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ñ –Ω–∞–±—É—Ç–æ–≥–æ –¥–æ—Å–≤—ñ–¥—É. –ú–µ—Ç–∞ –∞–≥–µ–Ω—Ç–∞, –≥—Ä—É–±–æ –∫–∞–∂—É—á–∏,"--- –º–∞–∫—Å–∏–º—ñ–∑—É–≤–∞—Ç–∏ —Å—É–∫—É–ø–Ω—É –≤–∏–Ω–∞–≥–æ—Ä–æ–¥—É, —è–∫—É –≤—ñ–Ω –æ—Ç—Ä–∏–º—É—î –ø—Ä–æ—Ç—è–≥–æ–º –±–∞–≥–∞—Ç—å–æ—Ö —Ö–æ–¥—ñ–≤. 

\subsection{–ü—Ä–∏–±—É—Ç–æ–∫ (return)}

–í–∏–∑–Ω–∞—á–∏–º–æ –±—ñ–ª—å—à —Ç–æ—á–Ω–æ –º–µ—Ç—É –∞–≥–µ–Ω—Ç–∞. –ù–µ—Ö–∞–π, –ø–æ—á–∏–Ω–∞—é—á–∏ –∑ —á–∞—Å—É $t$, –∞–≥–µ–Ω—Ç –æ—Ç—Ä–∏–º—É—î –ø–æ—Å–ª—ñ–¥–æ–≤–Ω—ñ—Å—Ç—å –≤–∏–Ω–∞–≥–æ—Ä–æ–¥ $r_{t+1}, r_{t+2}, r_{t+3}, \ldots$. –¢–æ–¥—ñ –æ—Å–Ω–æ–≤–Ω–æ—é –º–µ—Ç–æ—é –∞–≥–µ–Ω—Ç–∞ –±—É–¥–µ –º–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—è \emph{–æ—á—ñ–∫—É–≤–∞–Ω–æ–≥–æ –ø—Ä–∏–±—É—Ç–∫—É (expected return)} $R_t$, –¥–µ $R_t$ "--- —Ü–µ –ø–µ–≤–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –≤—ñ–¥ –ø–æ—Å–ª—ñ–¥–æ–≤–Ω–æ—Å—Ç—ñ –≤–∏–Ω–∞–≥–æ—Ä–æ–¥. –í –Ω–∞–π–ø—Ä–æ—Å—Ç—ñ—à–æ–º—É –≤–∏–ø–∞–¥–∫—É –ø—Ä–∏–±—É—Ç–æ–∫ —î —Å—É–º–æ—é —É—Å—ñ—Ö –≤–∏–Ω–∞–≥–æ—Ä–æ–¥:

\begin{equation}
R_t = r_{t+1} + r_{t+2} + \cdots + r_T,
\end{equation}
–¥–µ $T$ "--- –∫—ñ–Ω—Ü–µ–≤–∏–π –º–æ–º–µ–Ω—Ç —á–∞—Å—É. –¢–∞–∫–∏–π –ø—ñ–¥—Ö—ñ–¥ –¥–æ –≤–∏–∑–Ω–∞—á–µ–Ω–Ω—è –ø—Ä–∏–±—É—Ç–∫—É –ø—ñ–¥—Ö–æ–¥–∏—Ç—å —Ç–∞–º, –¥–µ –º–æ–∂–Ω–∞ –ø—Ä–∏—Ä–æ–¥–Ω–æ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ —Ñ—ñ–Ω–∞–ª—å–Ω–∏–π –º–æ–º–µ–Ω—Ç —á–∞—Å—É, —Ç–æ–±—Ç–æ –≤ —Ç–∞–∫–∏—Ö —Å–∏—Å—Ç–µ–º–∞—Ö, –≤ —è–∫–∏—Ö –≤–∑–∞—î–º–æ–¥—ñ—è –∞–≥–µ–Ω—Ç–∞ —ñ —Å–µ—Ä–µ–¥–æ–≤–∏—â–∞ –ø—Ä–∏—Ä–æ–¥–Ω—å–æ —Ä–æ–∑–±–∏–≤–∞—î—Ç—å—Å—è –Ω–∞ –æ–∫—Ä–µ–º—ñ –ø—ñ–¥–ø–æ—Å–ª—ñ–¥–æ–≤–Ω–æ—Å—Ç—ñ (\emph{–µ–ø—ñ–∑–æ–¥–∏}), –Ω–∞–ø—Ä–∏–∫–ª–∞–¥, –ø–∞—Ä—Ç—ñ—ó –≤ –≥—Ä—ñ, –ø—Ä–æ—Ö–æ–¥–∏ –ø–æ –ª–∞–±—ñ—Ä–∏–Ω—Ç—É –∞–±–æ –±—É–¥—å-—è–∫–∏–π —ñ–Ω—à–∏–π –≤–∏–¥ –ø–æ–≤—Ç–æ—Ä—é–≤–∞–Ω–∏—Ö –≤–∑–∞—î–º–æ–¥—ñ–π.

–ö–æ–∂–µ–Ω –µ–ø—ñ–∑–æ–¥ –∑–∞–∫—ñ–Ω—á—É—î—Ç—å—Å—è —É \emph{–∫—ñ–Ω—Ü–µ–≤–æ–º—É, —Ç–µ—Ä–º—ñ–Ω–∞–ª—å–Ω–æ–º—É —Å—Ç–∞–Ω—ñ (terminal state)}, –ø—ñ—Å–ª—è —è–∫–æ–≥–æ –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è –≤—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–Ω—è —Å—Ç–∞–Ω—É —Å–∏—Å—Ç–µ–º–∏ –≤ –ø–µ–≤–Ω–∏–π –ø–æ—á–∞—Ç–∫–æ–≤–∏–π —Å—Ç–∞–Ω –∑ –º–Ω–æ–∂–∏–Ω–∏ –º–æ–∂–ª–∏–≤–∏—Ö –ø–æ—á–∞—Ç–∫–æ–≤–∏—Ö —Å—Ç–∞–Ω—ñ–≤ —ñ —Ä–æ–∑–ø–æ—á–∏–Ω–∞—î—Ç—å—Å—è –Ω–∞—Å—Ç—É–ø–Ω–∏–π –µ–ø—ñ–∑–æ–¥. –ó–∞–¥–∞—á—ñ, –≤ —è–∫–∏—Ö –≤–∑–∞—î–º–æ–¥—ñ—è –∞–≥–µ–Ω—Ç–∞ —ñ —Å–µ—Ä–µ–¥–æ–≤–∏—â–∞ –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è —É –≤–∏–≥–ª—è–¥—ñ –ø–æ—Å–ª—ñ–¥–æ–≤–Ω–æ—Å—Ç—ñ –µ–ø—ñ–∑–æ–¥—ñ–≤, –Ω–∞–∑–∏–≤–∞—é—Ç—å—Å—è \emph{–µ–ø—ñ–∑–æ–¥–∏—á–Ω–∏–º–∏ –∑–∞–¥–∞—á–∞–º–∏ (episodic tasks)}. –í –µ–ø—ñ–∑–æ–¥–∏—á–Ω–∏—Ö –∑–∞–¥–∞—á–∞—Ö —ñ–Ω–∫–æ–ª–∏ —Ä–æ–∑–º–µ–∂–æ–≤—É—é—Ç—å –º–Ω–æ–∂–∏–Ω—É —É—Å—ñ—Ö –Ω–µ—Ç–µ—Ä–º—ñ–Ω–∞–ª—å–Ω–∏—Ö —Å—Ç–∞–Ω—ñ–≤ —Å–∏—Å—Ç–µ–º–∏ $\mathcal{S}$ —Ç–∞ –º–Ω–æ–∂–∏–Ω—É —É—Å—ñ—Ö —Å—Ç–∞–Ω—ñ–≤, –≤–∫–ª—é—á–Ω–æ –∑ —Ç–µ—Ä–º—ñ–Ω–∞–ª—å–Ω–∏–º–∏ $\mathcal{S^+}$. 

–ó —ñ–Ω—à–æ–≥–æ –±–æ–∫—É, –Ω–µ –∑–∞–≤–∂–¥–∏ –º–æ–∂–Ω–∞ –∑–¥—ñ–π—Å–Ω–∏—Ç–∏ –ø—Ä–∏—Ä–æ–¥–Ω—ñ–π –ø–æ–¥—ñ–ª –Ω–∞ –µ–ø—ñ–∑–æ–¥–∏, –Ω–∞—Ç–æ–º—ñ—Å—Ç—å –≤–∑–∞—î–º–æ–¥—ñ—è –≤—ñ–¥–±—É–≤–∞—î—Ç—å –ø–æ—Å—Ç—ñ–π–Ω–æ –±–µ–∑ –∑–∞–∫—ñ–Ω—á–µ–Ω–Ω—è. –ü—Ä–∏–∫–ª–∞–¥–æ–º —Ç–∞–∫–æ—ó –∑–∞–¥–∞—á—ñ –º–æ–∂–µ –±—É—Ç–∏ –¥—ñ—è–ª—å–Ω—ñ—Å—Ç—å —Ä–æ–±–æ—Ç–∞ –∑ –≤–µ–ª–∏–∫–æ—é —Ç—Ä–∏–≤–∞–ª—ñ—Å—Ç—é –∂–∏—Ç—Ç—î–¥—ñ—è–ª—å–Ω–æ—Å—Ç—ñ. –¢–∞–∫—ñ –∑–∞–¥–∞—á—ñ –Ω–∞–∑–≤–µ–º–æ \emph{–Ω–µ–ø–µ—Ä–µ—Ä–≤–Ω–∏–º–∏ –∑–∞–¥–∞—á–∞–º–∏ (continuing tasks)}. –í —Ç–∞–∫–æ–º—É –≤–∏–ø–∞–¥–∫—É –∫—ñ–Ω—Ü–µ–≤–∏–π –º–æ–º–µ–Ω—Ç —á–∞—Å—É $T=\infty$, –∞ —Å–∞–º –ø—Ä–∏–±—É—Ç–æ–∫, —è–∫–∏–π –º–∏ –Ω–∞–º–∞–≥–∞—î–º–æ—Å—è –º–∞–∫—Å–∏–º—ñ–∑—É–≤–∞—Ç–∏ –º–æ–∂–µ –ª–µ–≥–∫–æ –ø—Ä—è–º—É–≤–∞—Ç–∏ –¥–æ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–æ—Å—Ç—ñ (–Ω–∞–ø—Ä–∏–∫–ª–∞–¥, —è–∫—â–æ –Ω–∞ –∫–æ–∂–Ω–æ–º—É –∫—Ä–æ—Ü—ñ $r_t=1$). –©–æ–± –ø–æ–¥–æ–ª–∞—Ç–∏ —Ü—ñ —Ç—Ä—É–¥–Ω–æ—â—ñ, –≤–≤–µ–¥–µ–º–æ –¥–µ—â–æ —ñ–Ω—à–µ –≤–∏–∑–Ω–∞—á–µ–Ω–Ω—è –ø—Ä–∏–±—É—Ç–∫—É.

– –æ–∑–≥–ª—è–Ω–µ–º–æ –¥–æ–¥–∞—Ç–∫–æ–≤—É –∫–æ–Ω—Ü–µ–ø—Ü—ñ—é "--- \emph{–¥–∏—Å–∫–æ–Ω—Ç—É–≤–∞–Ω–Ω—è (discounting)}. –ó–≥—ñ–¥–Ω–æ –∑ –Ω–µ—é, –∞–≥–µ–Ω—Ç –Ω–∞–º–∞–≥–∞—î—Ç—Å—å—è –≤–∏–±–∏—Ä–∞—Ç–∏ —Ç–∞–∫—ñ –¥—ñ—ó, —â–æ–± –º–∞–∫—Å–∏–º—ñ–∑—É–≤–∞—Ç–∏ —Å—É–º—É –¥–∏—Å–∫–æ–Ω—Ç–æ–≤–∞–Ω–∏—Ö –≤–∏–Ω–∞–≥–æ—Ä–æ–¥, —è–∫—ñ –≤—ñ–Ω –æ—Ç—Ä–∏–º–∞—î –≤ –º–∞–π–±—É—Ç–Ω—å–æ–º—É. –ó–æ–∫—Ä–µ–º–∞, –∞–≥–µ–Ω—Ç –≤–∏–±–∏—Ä–∞—î —Ç–∞–∫—É –¥—ñ—é $a_t$, —è–∫–∞ –º–∞–∫—Å–∏–º—ñ–∑—É—î –æ—á—ñ–∫—É–≤–∞–Ω–∏–π \emph{–¥–∏—Å–∫–æ–Ω—Ç–æ–≤–∞–Ω–∏–π –ø—Ä–∏–±—É—Ç–æ–∫ (discounted reward)}:

\begin{equation}
R_t = r_{t+1} + \gamma r_{t+2} + \gamma^2 r_{t+3} + \cdots = \sum_{k=0}^{\infty}\gamma^k r_{t+k+1},
\end{equation}
–¥–µ $\gamma$ "--- —Ü–µ –ø–∞—Ä–∞–º–µ—Ç—Ä, —è–∫–∏–π –Ω–∞–∑–∏–≤–∞—î—Ç—å—Å—è \emph{–¥–∏—Å–∫–æ–Ω—Ç–Ω–æ—é —Å—Ç–∞–≤–∫–æ—é (discount rate)}, $0\leq\gamma\leq 1$.

–î–∏—Å–∫–æ–Ω—Ç–Ω–∞ —Å—Ç–∞–≤–∫–∞ –≤–∏–∑–Ω–∞—á–∞—î –ø–æ—Ç–æ—á–Ω—É –≤–∞–≥—É –º–∞–π–±—É—Ç–Ω—å–æ—ó –≤–∏–Ω–∞–≥–æ—Ä–æ–¥–∏: –≤–∏–Ω–∞–≥–æ—Ä–æ–¥–∞, –æ—Ç—Ä–∏–º–∞–Ω–∞ —á–µ—Ä–µ–∑ $k$ –∫—Ä–æ–∫—ñ–≤ —É –º–∞–π–±—É—Ç–Ω—å–æ–º—É, –º–∞—î –≤–∞–≥—É –ª–∏—à–µ $\gamma^k$, –Ω–∞–ø—Ä–æ—Ç–∏–≤–∞–≥—É –æ–¥–∏–Ω–∏—á–Ω—ñ–π –≤–∞–∑—ñ, —è–∫—É –± –≤–æ–Ω–∞ –º–∞–ª–∞, —è–∫–±–∏ –±—É–ª–∞ –æ—Ç—Ä–∏–º–∞–Ω–∞ –Ω–∞ –ø–æ—Ç–æ—á–Ω–æ–º—É –∫—Ä–æ—Ü—ñ. –Ø–∫—â–æ $\gamma<1$, —Ç–æ —Ü—è –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–∞ —Å—É–º–∞ –º–∞—î —Å–∫—ñ–Ω—á–µ–Ω–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è, —è–∫—â–æ –ø–æ—Å–ª—ñ–¥–æ–≤–Ω—ñ—Å—Ç—å $\left\{r_k\right\}$ "--- –æ–±–º–µ–∂–µ–Ω–∞. –Ø–∫—â–æ $\gamma = 0$, —Ç–æ –∞–≥–µ–Ω—Ç —î ``–Ω–µ–¥–∞–ª–µ–∫–æ–≥–ª—è–¥–Ω–∏–º'' —ñ –±—É–¥–µ –¥–±–∞—Ç–∏ –ª–∏—à–µ –ø—Ä–æ –º–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—é –º–∏—Ç—Ç—î–≤–æ—ó –≤–∏–Ω–∞–≥–æ—Ä–æ–¥–∏. –ô–æ–≥–æ –∑–∞–¥–∞—á–µ—é –≤ —Ç–∞–∫–æ–º—É –≤–∏–ø–∞–¥–∫—É –±—É–¥–µ –≤–∏–±—ñ—Ä —Ç–∞–∫–æ—ó –¥—ñ—ó $a_t$, —â–æ–± –º–∞–∫—Å–∏–º—ñ–∑—É–≤–∞—Ç–∏ –ª–∏—à–µ –≤–∏–Ω–∞–≥–æ—Ä–æ–¥—É $r_{t+1}$. –Ø–∫—â–æ –¥—ñ—è –∞–≥–µ–Ω—Ç–∞ –Ω–µ –∑–¥—ñ–π—Å–Ω—é—î –≤–ø–ª–∏–≤ –Ω–∞ –º–∞–π–±—É—Ç–Ω—ñ –≤–∏–Ω–∞–≥–æ—Ä–æ–¥–∏, —Ç–æ —Ç–∞–∫–∏–π –ø—ñ–¥—Ö—ñ–¥ –¥—ñ–π—Å–Ω–æ –¥–∞—Å—Ç—å –∑–º–æ–≥—É –≤–∏–≤—á–∏—Ç–∏ –æ–ø—Ç–∏–º–∞–ª—å–Ω—É —Å—Ç—Ä–∞—Ç–µ–≥—ñ—é, –æ–¥–Ω–∞–∫ –≤ –±—ñ–ª—å—à–æ—Å—Ç—ñ –≤–∏–ø–∞–¥–∫—ñ–≤ —Ü–µ –Ω–µ —Ç–∞–∫, —Ç–æ–º—É –º–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—è –ª–∏—à–µ –º–∏—Ç—Ç—î–≤–æ—ó –Ω–∞–≥–æ—Ä–æ–¥–∏ –º–æ–∂–µ –ø—Ä–∏–∑–≤–µ—Å—Ç–∏ –¥–æ —Ç–æ–≥–æ, —â–æ –±—É–¥–µ –≤—Ç—Ä–∞—á–µ–Ω–æ –Ω–∞–≥–æ–¥—É ``–∑–∞—Ä–æ–±–∏—Ç–∏'' –±—ñ–ª—å—à–µ –≤ –º–∞–π–±—É—Ç–Ω—å–æ–º—É, –∑–¥—ñ–π—Å–Ω–∏–≤—à–∏ –Ω–∞ –¥–∞–Ω–æ–º—É –∫—Ä–æ—Ü—ñ –¥—ñ—é, —è–∫–∞ –ø—Ä–∏–Ω–µ—Å–µ –Ω–µ –º–∞–∫—Å–∏–º–∞–ª—å–Ω—É –º–∏—Ç—Ç—î–≤—É –≤–∏–Ω–∞–≥–æ—Ä–æ–¥—É. –ó –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è–º $\gamma$ –¥–æ 1, –∞–≥–µ–Ω—Ç –±—É–¥–µ –Ω–∞–¥–∞–≤–∞—Ç–∏ –º–∞–π–±—É—Ç–Ω—ñ–º –≤–∏–Ω–∞–≥–æ—Ä–æ–¥–∞–º –≤—Å–µ –±—ñ–ª—å—à–æ–≥–æ –∑–Ω–∞—á–µ–Ω–Ω—è: –∞–≥–µ–Ω—Ç —Å—Ç–∞—î –±—ñ–ª—å—à ``–¥–∞–ª–µ–∫–æ–≥–ª—è–¥–Ω–∏–º''.

\subsection{–¶—ñ–Ω–Ω—ñ—Å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è (value function)}

–ü—Ä–∞–∫—Ç–∏—á–Ω–æ –≤—Å—ñ –∞–ª–≥–æ—Ä–∏—Ç–º–∏ –Ω–∞–≤—á–∞–Ω–Ω—è –∑ –ø—ñ–¥—Å–∏–ª–µ–Ω–Ω—è–º –±–∞–∑—É—é—Ç—å—Å—è –Ω–∞ –æ—Ü—ñ–Ω—Ü—ñ \emph{—Ü—ñ–Ω–Ω—ñ—Å–Ω–æ—ó —Ñ—É–Ω–∫—Ü—ñ—ó (value function)} "--- —Ñ—É–Ω–∫—Ü—ñ—ó —Å—Ç–∞–Ω—ñ–≤ —Å–∏—Å—Ç–µ–º–∏ (–∞–±–æ –ø–∞—Ä–∏ —Å—Ç–∞–Ω-–¥—ñ—è), —â–æ –æ—Ü—ñ–Ω—é—î \emph{–Ω–∞—Å–∫—ñ–ª—å–∫–∏ –¥–æ–±—Ä–µ} –¥–ª—è –∞–≥–µ–Ω—Ç–∞ –ø–µ—Ä–µ–±—É–≤–∞—Ç–∏ –≤ –∑–∞–¥–∞–Ω–æ–º—É —Å—Ç–∞–Ω—ñ (–∞–±–æ –Ω–∞—Å–∫—ñ–ª—å–∫–∏ –¥–æ–±—Ä–µ –∑–¥—ñ–π—Å–Ω–∏—Ç–∏ –∑–∞–¥–∞–Ω—É –¥—ñ—é, –ø–µ—Ä–µ–±—É–≤–∞—é—á–∏ –≤ –∑–∞–¥–∞–Ω–æ–º—É —Å—Ç–∞–Ω—ñ). –§—Ä–∞–∑–∞ ``–Ω–∞—Å–∫—ñ–ª—å–∫–∏ –¥–æ–±—Ä–µ'' –≤–∏–∑–Ω–∞—á–∞—î—Ç—å—Å—è –≤ —Ç–µ—Ä–º—ñ–Ω–∞—Ö –º–∞–π–±—É—Ç–Ω—ñ—Ö –≤–∏–Ω–∞–≥–æ—Ä–æ–¥ –∞–±–æ,—â–æ —î —Ç–æ—á–Ω—ñ—à–∏–º –≤–∏–∑–Ω–∞—á–µ–Ω–Ω—è–º, –≤ —Ç–µ—Ä–º—ñ–Ω–∞—Ö –æ—á—ñ–∫—É–≤–∞–Ω–æ–≥–æ –º–∞–π–±—É—Ç–Ω—å–æ–≥–æ –ø—Ä–∏–±—É—Ç–∫—É. –û—á–µ–≤–∏–¥–Ω–æ, —â–æ –º–∞–π–±—É—Ç–Ω—ñ –≤–∏–Ω–∞–≥–æ—Ä–æ–¥–∏ –∑–∞–ª–µ–∂–∞—Ç—å –≤—ñ–¥ —Ç–æ–≥–æ, —è–∫—ñ –¥—ñ—ó –∑–¥—ñ–π—Å–Ω—é–≤–∞—Ç–∏, —Ç–æ–±—Ç–æ –≤—ñ–¥ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó, —Ç–æ–º—É —Ü—ñ–Ω–Ω—ñ—Å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –≤–∏–∑–Ω–∞—á–∞—î—Ç—å—Å—è –∑ –≤—Ä–∞—Ö—É–≤–∞–Ω–Ω—è–º –ø–µ–≤–Ω–æ—ó —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó. 

–ù–∞–≥–∞–¥–∞—î–º–æ, —â–æ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—è $\pi$ "--- —Ü–µ –≤—ñ–¥–æ–±—Ä–∞–∂–µ–Ω–Ω—è –∫–æ–∂–Ω–æ–≥–æ —Å—Ç–∞–Ω—É $s \in \mathcal{S}$ —Ç–∞ –∫–æ–∂–Ω–æ—ó –¥–æ–ø—É—Å—Ç–∏–º–æ—ó –¥—ñ—ó $a \in \mathcal{A}(s)$ –Ω–∞ –π–º–æ–≤—ñ—Ä–Ω—ñ—Å—Ç—å $\pi(s,a)$ –∑–¥—ñ–π—Å–Ω–µ–Ω–Ω—è –¥—ñ—ó $a$, –±—É–¥—É—á–∏ –≤ —Å—Ç–∞–Ω—ñ $s$. \emph{–¶—ñ–Ω–Ω—ñ—Å—Ç—å} —Å—Ç–∞–Ω—É $s$ –∑–≥—ñ–¥–Ω–æ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó $\pi$, $V^\pi(s)$,"--- —Ü–µ –æ—á—ñ–∫—É–≤–∞–Ω–∏–π –ø—Ä–∏–±—É—Ç–æ–∫, —è–∫–∏–π –º–æ–∂–Ω–∞ –æ—Ç—Ä–∏–º–∞—Ç–∏, —è–∫—â–æ –ø–æ—á–∞—Ç–∏ –∑—ñ —Å—Ç–∞–Ω—É $s$ —ñ –¥—ñ—è—Ç–∏ –∑–≥—ñ–¥–Ω–æ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó $\pi$:
\begin{equation}
V^\pi(s) = E_\pi \Bigl\{ R_t \Big| s_t = s \Bigr \} = E_\pi \Biggl\{\sum_{k=0}^{\infty}\gamma^k r_{t+k+1} \Bigg| s_t=s \Biggr\},
\end{equation}
–¥–µ $E_\pi{}$ –ø–æ–∑–Ω–∞—á–∞—î –º–∞—Ç–µ–º–∞—Ç–∏—á–Ω–µ –æ—á—ñ–∫—É–≤–∞–Ω–Ω—è –≤ –≤–∏–ø–∞–¥–∫—É, —è–∫—â–æ –∞–≥–µ–Ω—Ç –¥—ñ—î –∑–≥—ñ–¥–Ω–æ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó $\pi$. –ü–æ—Ç—Ä—ñ–±–Ω–æ –≤—ñ–¥–º—ñ—Ç–∏—Ç–∏, —â–æ —Ü—ñ–Ω–Ω—ñ—Å—Ç—å —Ç–µ—Ä–º—ñ–Ω–∞–ª—å–Ω–æ–≥–æ —Å—Ç–∞–Ω—É –∑–∞–≤–∂–¥–∏ —Ä—ñ–≤–Ω–∞ –Ω—É–ª—é. –ù–∞–∑–≤–µ–º–æ —Ñ—É–Ω–∫—Ü—ñ—é $V^\pi$ \emph{—Ü—ñ–Ω–Ω—ñ—Å–Ω–æ—é —Ñ—É–Ω–∫—Ü—ñ—î—é —Å—Ç–∞–Ω—ñ–≤ (state-value function) –¥–ª—è —Å—Ç–∞—Ç–µ–≥—ñ—ó $\pi$}.

–ê–Ω–∞–ª–æ–≥—ñ—á–Ω–æ, –≤–∏–∑–Ω–∞—á–∏–º–æ —Ü—ñ–Ω–Ω—ñ—Å—Ç—å –∑–¥—ñ–π—Å–Ω–µ–Ω–Ω—è –¥—ñ—ó $a$ –≤ —Å—Ç–∞–Ω—ñ $s$ –∑–≥—ñ–¥–Ω–æ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó $\pi$, $Q^\pi(s,a)$, —è–∫ –æ—á—ñ–∫—É–≤–∞–Ω–∏–π –ø—Ä–∏–±—É—Ç–æ–∫, –æ—Ç—Ä–∏–º–∞–Ω–∏–π –≤ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ñ –ø–µ—Ä–µ–±—É–≤–∞–Ω–Ω—è –≤ —Å—Ç–∞–Ω—ñ $s$, –∑–¥—ñ–π—Å–Ω–µ–Ω–Ω—è –¥—ñ—ó $a$ —Ç–∞ –ø–æ–¥–∞–ª—å—à–æ–º—É —Å–ª—ñ–¥—É–≤–∞–Ω–Ω—ñ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó $\pi$:
\begin{equation}
Q^\pi(s,a)=E_\pi \Bigl\{ R_t \Big| s_t=s, a_t=a \Bigr\} = E_\pi \Biggl\{\sum_{k=0}^{\infty}\gamma^k r_{t+k+1} \Bigg| s_t=s,a_t=a \Biggr\}.
\end{equation}
–ù–∞–∑–≤–µ–º–æ —Ñ—É–Ω–∫—Ü—ñ—é $Q^\pi$ \emph{—Ü—ñ–Ω–Ω—ñ—Å–Ω–æ—é —Ñ—É–Ω–∫—Ü—ñ—î—é –¥—ñ–π (action-value function) –¥–ª—è —Å—Ç–∞—Ç–µ–≥—ñ—ó $\pi$}.

–¶—ñ–Ω–Ω—ñ—Å–Ω—ñ —Ñ—É–Ω–∫—Ü—ñ—ó $V^\pi$ —Ç–∞ $Q^\pi$ –º–æ–∂—É—Ç—å –±—É—Ç–∏ –æ—Ü—ñ–Ω–µ–Ω—ñ –∑ –¥–æ—Å–≤—ñ–¥—É. –ù–∞–ø—Ä–∏–∫–ª–∞–¥, —è–∫—â–æ –∞–≥–µ–Ω—Ç –ø—Ä–∏—Ç—Ä–∏–º—É—î—Ç—å—Å—è —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó $\pi$ —ñ –ø—ñ–¥—Ç—Ä–∏–º—É—î —Å–µ—Ä–µ–¥–Ω—î –∑–Ω–∞—á–µ–Ω–Ω—è –≤—Å—ñ—Ö –ø—Ä–∏–±—É—Ç–∫—ñ–≤, —è–∫—ñ –±—É–ª–∏ —Ä–µ–∞–ª—å–Ω–æ –æ—Ç—Ä–∏–º–∞–Ω—ñ, –¥–ª—è –∫–æ–∂–Ω–æ–≥–æ —Å—Ç–∞–Ω—É, –≤ —è–∫–∏—Ö –ø–µ—Ä–µ–±—É–≤–∞–≤ –∞–≥–µ–Ω—Ç, —Ç–æ —Å–µ—Ä–µ–¥–Ω—î –∑–Ω–∞—á–µ–Ω–Ω—è –∑–±—ñ–∂–∏—Ç—å—Å—è –¥–æ –∑–Ω–∞—á–µ–Ω–Ω—è —Ü—ñ–Ω–Ω–æ—Å—Ç—ñ –¥–ª—è –¥–∞–Ω–æ–≥–æ —Å—Ç–∞–Ω—É $V^\pi(s)$ –∑–∞ —É–º–æ–≤–∏, —â–æ –∫—ñ–ª—å–∫—ñ—Å—Ç—å –ø–µ—Ä–µ–±—É–≤–∞–Ω—å –≤ –∫–æ–∂–Ω–æ–º—É –∑—ñ —Å—Ç–∞–Ω—ñ–≤ –ø—Ä—è–º—É—î –¥–æ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ—Å—Ç—ñ. –Ø–∫—â–æ –∑–±–µ—Ä—ñ–≥–∞—Ç–∏ —Å–µ—Ä–µ–¥–Ω—ñ –∑–Ω–∞—á–µ–Ω–Ω—è –æ–∫—Ä–µ–º–æ –¥–ª—è –∫–æ–∂–Ω–æ—ó –ø–∞—Ä–∏ —Å—Ç–∞–Ω-–¥—ñ—è, —Ç–æ —Ç–∞–∫–µ –∂ —Ç–≤–µ—Ä–¥–∂–µ–Ω–Ω—è –≤—ñ—Ä–Ω–µ —ñ –¥–ª—è —Ñ—É–Ω–∫—Ü—ñ—ó $Q^\pi$. –¢–∞–∫—ñ –º–µ—Ç–æ–¥–∏ –æ—Ü—ñ–Ω–∫–∏ —Ü—ñ–Ω–Ω—ñ—Å–Ω–∏—Ö —Ñ—É–Ω–∫—Ü—ñ–π –Ω–∞–∑–∏–≤–∞—é—Ç—å—Å—è \emph{–º–µ—Ç–æ–¥–∞–º–∏ –ú–æ–Ω—Ç–µ-–ö–∞—Ä–ª–æ}, —Ç–æ–º—É —â–æ –≤–æ–Ω–∏ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—Ç—å —É—Å–µ—Ä–µ–¥–Ω–µ–Ω–Ω—è –ø–æ –≤–∏–ø–∞–¥–∫–æ–≤–∏—Ö –ø—Ä–∏–∫–ª–∞–¥–∞—Ö —Ä–µ–∞–ª—å–Ω–∏—Ö –ø—Ä–∏–±—É—Ç–∫—ñ–≤. –ü—Ä–æ—Ç–µ, —è–∫—â–æ –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Å—Ç–∞–Ω—ñ–≤ —Ç–∞ –º–æ–∂–ª–∏–≤–∏—Ö –¥—ñ–π –≤–µ–ª–∏–∫–∞, —Ü–µ —É–Ω–µ–º–æ–∂–ª–∏–≤–ª—é—î –∑–±–µ—Ä–µ–∂–µ–Ω–Ω—è —Å–µ—Ä–µ–¥–Ω—ñ—Ö –∑–Ω–∞—á–µ–Ω—å –¥–ª—è –∫–æ–∂–Ω–æ–≥–æ —Å—Ç–∞–Ω—É (–ø–∞—Ä–∏ —Å—Ç–∞–Ω-–¥—ñ—è). –í —Ç–∞–∫–æ–º—É –≤–∏–ø–∞–¥–∫—É –º–æ–∂–Ω–∞ –∑–∞—Å—Ç–æ—Å—É–≤–∞—Ç–∏ —ñ–Ω—à—ñ –º–µ—Ç–æ–¥–∏, –∑–æ–∫—Ä–µ–º–∞ —Ç–∞–∫—ñ, —â–æ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—Ç—å –∞–ø—Ä–æ–∫—Å–∏–º–∞—Ç–æ—Ä–∏ —Ñ—É–Ω–∫—Ü—ñ–π. –ú–∏ —Ä–æ–∑–≥–ª—è–Ω–µ–º–æ —ó—Ö –¥–µ—â–æ –ø—ñ–∑–Ω—ñ—à–µ.

–§—É–Ω–¥–∞–º–µ–Ω—Ç–∞–ª—å–Ω–∞ –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—å —Ü—ñ–Ω–Ω—ñ—Å–Ω–∏—Ö —Ñ—É–Ω–∫—Ü—ñ–π, —è–∫–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î—Ç—å—Å—è –≤ –Ω–∞–≤—á–∞–Ω–Ω—ñ –∑ –ø—ñ–¥—Å–∏–ª–µ–Ω–Ω—è–º —ñ –¥–∏–Ω–∞–º—ñ—á–Ω–æ–º—É –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—ñ,"--- —Ü–µ –ø–µ–≤–Ω–µ —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–µ —Å–ø—ñ–≤–≤—ñ–¥–Ω–æ—à–µ–Ω–Ω—è, —è–∫–æ–º—É –∑–∞–¥–æ–≤—ñ–ª—å–Ω—è—î —Ü—ñ–Ω–Ω—ñ—Å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è. –î–ª—è –±—É–¥—å-—è–∫–æ—ó —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó $\pi$ —Ç–∞ –±—É–¥—å-—è–∫–æ–≥–æ —Å—Ç–∞–Ω—É $s$, –Ω–∞—Å—Ç—É–ø–Ω–µ —Å–ø—ñ–≤–≤—ñ–¥–Ω–æ—à–µ–Ω–Ω—è –∑–∞–≤–∂–¥–∏ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è:
\begin{eqnarray}
V^\pi(s) & = & E_\pi \Bigl\{R_t \Big| s_t = s \Bigr\} \nonumber\\
			& = & E_\pi \Biggl\{\sum_{k=0}^{\infty}\gamma^k r_{t+k+1} \Bigg| s_t = s \Biggr\} \nonumber\\
			& = & \sum_a \pi(s,a)\sum_{s'}\mathcal{P}^a_{ss'} \left[ \mathcal{R}^a_{ss'}+\gamma E_\pi \left\{\sum_{k=0}^{\infty}\gamma^k r_{t+k+2} \Bigg| s_{t+1} = s' \right\} \right] \nonumber\\
			& = & \sum_a \pi(s,a)\sum_{s'}\mathcal{P}^a_{ss'} \left[ \mathcal{R}^a_{ss'}+\gamma V^\pi(s') \right],
\label{Bellman_eq}
\end{eqnarray}
–¥–µ $\mathcal{P}^a_{ss'}$ "--- —Ü–µ –π–º–æ–≤—ñ—Ä–Ω—ñ—Å—Ç—å –ø–µ—Ä–µ—Ö–æ–¥—É (transition probability) –∑—ñ —Å—Ç–∞–Ω—É $s$ –≤ —Å—Ç–∞–Ω $s'$ –ø—Ä–∏ –≤–∏–∫–æ–Ω–∞–Ω–Ω—ñ –¥—ñ—ó $a$; $\mathcal{R}^a_{ss'}$ "--- –æ—á—ñ–∫—É–≤–∞–Ω–∞ –≤–∏–Ω–∞–≥–æ—Ä–æ–¥–∞, —è–∫–∞ –¥–∞—î—Ç—å—Å—è –∞–≥–µ–Ω—Ç—É –ø—Ä–∏ –ø–µ—Ä–µ—Ö–æ–¥—ñ –∑—ñ —Å—Ç–∞–Ω—É $s$ –≤ —Å—Ç–∞–Ω $s'$ –ø—Ä–∏ –≤–∏–∫–æ–Ω–∞–Ω–Ω—ñ –¥—ñ—ó $a$. –í –¥–∞–Ω—ñ–π —Ñ–æ—Ä–º—É–ª—ñ –Ω–µ—è–≤–Ω–æ —Ä–æ–∑—É–º—ñ—î—Ç—å—Å—è, —â–æ $s \in \mathcal{S}$, –¥—ñ—ó $a$ –±–µ—Ä—É—Ç—å—Å—è –∑ –º–Ω–æ–∂–∏–Ω–∏ $\mathcal{A}(s)$, –∞ $s' \in \mathcal{S}$ —É –≤–∏–ø–∞–¥–∫—É –Ω–µ–ø–µ—Ä–µ—Ä–≤–Ω–æ—ó –∑–∞–¥–∞—á—ñ –∞–±–æ $s' \in \mathcal{S^+}$ "--- —É –≤–∏–ø–∞–¥–∫—É –µ–ø—ñ–∑–æ–¥–∏—á–Ω–æ—ó –∑–∞–¥–∞—á—ñ.

– —ñ–≤–Ω—è–Ω–Ω—è (\ref{Bellman_eq}) –Ω–∞–∑–∏–≤–∞—î—Ç—å—Å—è \emph{—Ä—ñ–≤–Ω—è–Ω–Ω—è–º –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ—Å—Ç—ñ –ë–µ–ª–ª–º–∞–Ω–∞ –¥–ª—è $V^\pi$ (Bellman optimality equation for $V^\pi$)}. –í–æ–Ω–æ –≤—ñ–¥–æ–±—Ä–∞–∂–∞—î –∑–≤'—è–∑–æ–∫ –º—ñ–∂ —Ü—ñ–Ω–Ω—ñ—Å—Ç—é —Å—Ç–∞–Ω—É $s$ —Ç–∞ —Ü—ñ–Ω–Ω–æ—Å—Ç—è–º–∏ —Å—Ç–∞–Ω—ñ–≤ $s'$, –≤ —è–∫—ñ –º–æ–∂–Ω–∞ –ø–æ—Ç—Ä–∞–ø–∏—Ç–∏ –∑ $s$. –§—É–Ω–∫—Ü—ñ—è $V^\pi$ "--- —î–¥–∏–Ω–∏–π —Ä–æ–∑–≤'—è–∑–æ–∫ —Ä—ñ–≤–Ω—è–Ω–Ω—è –ë–µ–ª–ª–º–∞–Ω–∞. –í—ñ–¥–º—ñ—Ç–∏–º–æ —Ç–∞–∫–æ–∂, —â–æ —Ü–µ —Ä—ñ–≤–Ω—è–Ω–Ω—è "--- –æ—Å–Ω–æ–≤–∞ –¥–ª—è –±–∞–≥–∞—Ç—å–æ—Ö –º–µ—Ç–æ–¥—ñ–≤ –æ–±—á–∏—Å–ª–µ–Ω–Ω—è, –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è —Ç–∞ –≤–∏–≤—á–µ–Ω–Ω—è $V^\pi$.

– –æ–∑–≤'—è–∑—É–≤–∞–Ω–Ω—è –∑–∞–¥–∞—á—ñ –Ω–∞–≤—á–∞–Ω–Ω—è –∑ –ø—ñ–¥—Å–∏–ª–µ–Ω–Ω—è–º –æ–∑–Ω–∞—á–∞—î, –∑–≥—Ä—É–±—à–∞, –∑–Ω–∞—Ö–æ–¥–∂–µ–Ω–Ω—è —Ç–∞–∫–æ—ó —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó, —è–∫–∞ –± –¥–∞–≤–∞–ª–∞ –≤–µ–ª–∏–∫–∏–π –ø—Ä–∏–±—É—Ç–æ–∫ –≤ –¥–æ–≤–≥–æ—Å—Ç—Ä–æ–∫–æ–≤—ñ–π –ø–µ—Ä—Å–ø–µ–∫—Ç–∏–≤—ñ. –°–ø—Ä–æ–±—É—î–º–æ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ –æ–ø—Ç–∏–º–∞–ª—å–Ω—ñ—Å—Ç—å —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó –Ω–∞—Å—Ç—É–ø–Ω–∏–º —á–∏–Ω–æ–º. –¶—ñ–Ω–Ω—ñ—Å–Ω—ñ —Ñ—É–Ω–∫—Ü—ñ—ó –≤–∏–∑–Ω–∞—á–∞—é—Ç—å —á–∞—Å—Ç–∫–æ–≤–∏–π –ø–æ—Ä—è–¥–æ–∫ –Ω–∞ –º–Ω–æ–∂–∏–Ω—ñ —É—Å—ñ—Ö —Å—Ç—Ä–∞—Ç–µ–≥—ñ–π. –°—Ç—Ä–∞—Ç–µ–≥—ñ—é $\pi$ –Ω–∞–∑–∏–≤–∞—Ç–∏–º–µ–º–æ \emph{–∫—Ä–∞—â–æ—é}, –Ω—ñ–∂ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—è $\pi '$, —è–∫—â–æ —ó—ó –æ—á—ñ–∫—É–≤–∞–Ω–∏–π –ø—Ä–∏–±—É—Ç–æ–∫ –±—ñ–ª—å—à–∏–π –∞–±–æ —Ä—ñ–≤–Ω–∏–π –≤—ñ–¥ —Ç–∞–∫–æ–≥–æ –∂ –¥–ª—è —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó $\pi '$ –¥–ª—è —É—Å—ñ—Ö —Å—Ç–∞–Ω—ñ–≤ $s \in \mathcal{S}$:
\begin{equation}

\end{equation}

–ó–∞–≤–∂–¥–∏ —ñ—Å–Ω—É—î —Ö–æ—á–∞ –± –æ–¥–Ω–∞ —Ç–∞–∫–∞ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—è, —è–∫–∞ –∫—Ä–∞—â–∞ –∞–±–æ —Ä—ñ–≤–Ω–∞ –∑–∞ –≤—Å—ñ—Ö —ñ–Ω—à–∏—Ö. –¢–∞–∫–∞ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—è –Ω–∞–∑–∏–≤–∞—î—Ç—å—Å—è \emph{–æ–ø—Ç–∏–º–∞–ª—å–Ω–æ—é}. –•–æ—á–∞ –æ–ø—Ç–∏–º–∞–ª—å–Ω–∏—Ö —Å—Ç—Ä–∞—Ç–µ–≥—ñ–π –º–æ–∂–µ –±—É—Ç–∏ –¥–µ–∫—ñ–ª—å–∫–∞, –≤—Å—ñ—Ö —ó—Ö –±—É–¥–µ–º–æ –ø–æ–∑–Ω–∞—á–∞—Ç–∏ —è–∫ $\pi^*$. –£—Å—ñ –æ–ø—Ç–∏–º–∞–ª—å–Ω—ñ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó —Ä–æ–∑–¥—ñ–ª—è—é—Ç—å –æ–¥–Ω—É —ñ —Ç—É –∂ —Ü—ñ–Ω–Ω—ñ—Å–Ω—É —Ñ—É–Ω–∫—Ü—ñ—é —Å—Ç–∞–Ω—ñ–≤, —è–∫–∞ –Ω–∞–∑–∏–≤–∞—î—Ç—å—Å—è \emph{–æ–ø—Ç–∏–º–∞–ª—å–Ω–æ—é —Ü—ñ–Ω–Ω—ñ—Å–Ω–æ—é —Ñ—É–Ω–∫—Ü—ñ—î—é —Å—Ç–∞–Ω—ñ–≤} $V^*$:
\begin{equation}

\end{equation}

–û–ø—Ç–∏–º–∞–ª—å–Ω—ñ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó —Ç–∞–∫–æ–∂ —Ä–æ–∑–¥—ñ–ª—è—é—Ç—å –æ–¥–Ω—É —ñ —Ç—É –∂ \emph{—Ü—ñ–Ω–Ω—ñ—Å–Ω—É —Ñ—É–Ω–∫—Ü—ñ—é –¥—ñ–π} $Q^*$:

\end{equation}

\subsubsection{Soft —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó}

–î–ª—è soft —Å—Ç—Ä–∞—Ç–µ–≥—ñ–π, –π–º–æ–≤—ñ—Ä–Ω—ñ—Å—Ç—å $\pi(s,a)>0,\ \forall \pi, \forall s \in \mathcal{S}, a \in \mathcal{A}(s)$.

–û–¥–Ω–∏–º –∑ –Ω–∞–π–ø—Ä–æ—Å—Ç—ñ—à–∏—Ö –≤–∏–ø–∞–¥–∫—ñ–≤ soft —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó —î $\varepsilon$-–∂–∞–¥—ñ–±–Ω–∞ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—è. –í —Ç–∞–∫—ñ–π —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó –≤ –æ—Å–Ω–æ–≤–Ω–æ–º—É –≤–∏–±–∏—Ä–∞—î—Ç—å—Å—è –¥—ñ—è, —è–∫–∞ –¥–∞—î –ø–æ—Ç–æ—á–Ω–∏–π –º–∞–∫—Å–∏–º—É–º —Ü—ñ–Ω–Ω—ñ—Å–Ω–æ—ó —Ñ—É–Ω–∫—Ü—ñ—ó, –ø—Ä–æ—Ç–µ —ñ–Ω–∫–æ–ª–∏, –∑ –π–º–æ–≤—ñ—Ä–Ω—ñ—Å—Ç—é $\varepsilon$ –ø—Ä–∏–π–º–∞—î—Ç—å—Å—è —Ä—ñ—à–µ–Ω–Ω—è –ø—Ä–æ –∑–¥—ñ–π—Å–Ω–µ–Ω–Ω—è –¥–æ–≤—ñ–ª—å–Ω–æ—ó –≤–∏–ø–∞–¥–∫–æ–≤–æ—ó –¥–æ–ø—É—Å—Ç–∏–º–æ—ó –¥—ñ—ó. –¢–∞–∫–∏–º —á–∏–Ω–æ–º –±—É–¥—å-—è–∫–∞ –Ω–µ –∂–∞–¥—ñ–±–Ω–∞ –¥—ñ—è –æ—Ç—Ä–∏–º—É—î –π–º–æ–≤—ñ—Ä–Ω—ñ—Å—Ç—å –≤–∏–∫–æ–Ω–∞–Ω–Ω—è, —Ä—ñ–≤–Ω—É $\frac{\varepsilon}{|\mathcal{A}(s)|}$, –∞ –∂–∞–¥—ñ–±–Ω–∞ –¥—ñ—è, –≤—ñ–¥–ø–æ–≤—ñ–¥–Ω–æ, $ 1 - \varepsilon + \frac{\varepsilon}{|\mathcal{A}(s)|}$.

\subsection{–ù–∞–≤—á–∞–Ω–Ω—è –∑ —á–∞—Å–æ–≤–æ—é —Ä—ñ–∑–Ω–∏—Ü–µ—é (TD-–º–µ—Ç–æ–¥–∏)}

–Ü–¥–µ—è –º–µ—Ç–æ–¥—ñ–≤ \emph{–Ω–∞–≤—á–∞–Ω–Ω—è –∑ —á–∞—Å–æ–≤–æ—é —Ä—ñ–∑–Ω–∏—Ü–µ—é (temporal-difference, TD-–Ω–∞–≤—á–∞–Ω–Ω—è)} —î, –º–∞–±—É—Ç—å, —Ü–µ–Ω—Ç—Ä–∞–ª—å–Ω–æ—é —Ç–∞ –Ω–∞–π–±—ñ–ª—å—à –Ω–æ–≤–∞—Ç–æ—Ä—Å—å–∫–æ—é –¥–ª—è –≤—Å—å–æ–≥–æ –Ω–∞–≤—á–∞–Ω–Ω—è –∑ –ø—ñ–¥—Å–∏–ª–µ–Ω–Ω—è–º. TD-–Ω–∞–≤—á–∞–Ω–Ω—è "--- —Ü–µ –ø–æ—î–¥–Ω–∞–Ω–Ω—è —ñ–¥–µ–π –º–µ—Ç–æ–¥—ñ–≤ –ú–æ–Ω—Ç–µ-–ö–∞—Ä–ª–æ —Ç–∞ –¥–∏–Ω–∞–º—ñ—á–Ω–æ–≥–æ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—è. TD-–º–µ—Ç–æ–¥–∏ –º–æ–∂—É—Ç—å, —è–∫ —ñ –º–µ—Ç–æ–¥–∏ –ú–æ–Ω—Ç–µ-–ö–∞—Ä–ª–æ, –Ω–∞–≤—á–∞—Ç–∏—Å—è –ª–∏—à–µ –∑ –¥–æ—Å–≤—ñ–¥—É, –Ω–µ –ø–æ—Ç—Ä–µ–±—É—é—á–∏ –∑–Ω–∞–Ω–Ω—è –º–æ–¥–µ–ª—ñ –¥–∏–Ω–∞–º—ñ–∫–∏ —Å–µ—Ä–µ–¥–æ–≤–∏—â–∞. –ó —ñ–Ω—à–æ–≥–æ –±–æ–∫—É, —Ç–∞–∫ —Å–∞–º–æ —è–∫ —ñ –≤ –º–µ—Ç–æ–¥–∞—Ö –¥–∏–Ω–∞–º—ñ—á–Ω–æ–≥–æ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—è, TD-–º–µ—Ç–æ–¥–∏ –æ–Ω–æ–≤–ª—é—é—Ç—å —Å–≤–æ—ó –æ—Ü—ñ–Ω–∫–∏ —á–∞—Å—Ç–∫–æ–≤–æ –Ω–∞ –æ—Å–Ω–æ–≤—ñ —ñ–Ω—à–∏—Ö –æ—Ü—ñ–Ω–æ–∫, –Ω–µ —á–µ–∫–∞—é—á–∏ –∑–∞–∫—ñ–Ω—á–µ–Ω–Ω—è –µ–ø—ñ–∑–æ–¥—É (–≤–æ–Ω–∏ –∑–¥—ñ–π—Å–Ω—é—é—Ç—å bootstrap).

\subsubsection{TD-–ø—Ä–æ–≥–Ω–æ–∑—É–≤–∞–Ω–Ω—è}

–Ø–∫ TD, —Ç–∞–∫ —ñ –ú–æ–Ω—Ç–µ-–ö–∞—Ä–ª–æ –º–µ—Ç–æ–¥–∏ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—Ç—å –æ—Ç—Ä–∏–º–∞–Ω–∏–π –¥–æ—Å–≤—ñ–¥ –¥–ª—è —Ä–æ–∑–≤'—è–∑—É–≤–∞–Ω–Ω—è –∑–∞–¥–∞—á—ñ –ø—Ä–æ–≥–Ω–æ–∑—É–≤–∞–Ω–Ω—è. –ú–∞—é—á–∏ –ø–µ–≤–Ω–∏–π –¥–æ—Å–≤—ñ–¥, –æ—Ç—Ä–∏–º–∞–Ω–∏–π –≤ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ñ –ø—Ä–∏—Ç—Ä–∏–º—É–≤–∞–Ω–Ω—è —Å—Ç—Ä–∞—Ç–µ–≥—ñ—ó $\pi$, –æ–±–∏–¥–≤–∞ –º–µ—Ç–æ–¥–∏ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—Ç—å –π–æ–≥–æ –¥–ª—è –æ–Ω–æ–≤–ª–µ–Ω–Ω—è –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è $V$ –¥–æ $V^\pi$. –Ø–∫—â–æ –≤ –º–æ–º–µ–Ω—Ç —á–∞—Å—É $t$ –±—É–≤ –≤—ñ–¥–≤—ñ–¥–∞–Ω–∏–π –Ω–µ—Ç–µ—Ä–º—ñ–Ω–∞–ª—å–Ω–∏–π —Å—Ç–∞–Ω $s_t$, —Ç–æ –æ–±–∏–¥–≤–∞ –º–µ—Ç–æ–¥–∏ –æ–Ω–æ–≤–ª—è—Ç—å —Å–≤–æ—ó –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è $V(s_t)$, –±–∞–∑—É—é—á–∏—Å—å –Ω–∞ —Ç–æ–º—É, —â–æ –≤—ñ–¥–±—É–¥–µ—Ç—å—Å—è –≤ –Ω–∞—Å—Ç—É–ø–Ω—ñ –º–æ–º–µ–Ω—Ç–∏ —á–∞—Å—É. –¢–∞–∫, –º–µ—Ç–æ–¥ –ú–æ–Ω—Ç–µ-–ö–∞—Ä–ª–æ —á–µ–∫–∞—î –º–æ–º–µ–Ω—Ç—É, –∫–æ–ª–∏ —Å—Ç–∞–Ω–µ –≤—ñ–¥–æ–º–∏–º –ø—Ä–∏–±—É—Ç–æ–∫, –æ—Ç—Ä–∏–º–∞–Ω–∏–π –ø—ñ—Å–ª—è –≤—ñ–¥–≤—ñ–¥–∞–Ω–Ω—è –ø–æ—Ç–æ—á–Ω–æ–≥–æ —Å—Ç–∞–Ω—É, –∞ —Ç–æ–¥—ñ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î —Ü–µ–π –ø—Ä–∏–±—É—Ç–æ–∫ —è–∫ –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è –¥–ª—è $V(s_t)$. –ü—Ä–æ—Å—Ç–∏–π –∞–ª–≥–æ—Ä–∏—Ç–º –ú–æ–Ω—Ç–µ-–ö–∞—Ä–ª–æ –¥–ª—è –≤—Å—ñ—Ö –≤—ñ–∑–∏—Ç—ñ–≤ (every-visit Monte Carlo) –≤–∏–≥–ª—è–¥–∞—î –Ω–∞—Å—Ç—É–ø–Ω–∏–º —á–∏–Ω–æ–º: 
\begin{equation}
V(s_t) \leftarrow V(s_t) + \alpha\left[R_t - V(s_t)\right],
\end{equation}
–¥–µ $R_t$ "--- —Ä–µ–∞–ª—å–Ω–∏–π –ø—Ä–∏–±—É—Ç–æ–∫, –æ—Ç—Ä–∏–º–∞–Ω–∏–π –ø—Ä–æ—Ç—è–≥–æ–º –ø–æ—Ç–æ—á–Ω–æ–≥–æ –µ–ø—ñ–∑–æ–¥—É, –ø–æ—á–∏–Ω–∞—é—á–∏ –∑ –º–æ–º–µ–Ω—Ç—É —á–∞—Å—É $t$, $\alpha$ "--- —Å—Ç–∞–ª–∏–π –ø–∞—Ä–∞–º–µ—Ç—Ä –≤–µ–ª–∏—á–∏–Ω–∏ –∫—Ä–æ–∫—É. –¢–∞–∫–∏–π –∞–ª–≥–æ—Ä–∏—Ç–º –Ω–∞–∑–∏–≤–∞—Ç–∏–º–µ–º–æ \emph{constant-$\alpha$ MC}.

–í —Ç–æ–π —á–∞—Å, —è–∫ MC-–º–µ—Ç–æ–¥ —á–µ–∫–∞—î –¥–æ –∑–∞–∫—ñ–Ω—á–µ–Ω–Ω—è –µ–ø—ñ–∑–æ–¥—É, TD-–º–µ—Ç–æ–¥ –ø–æ–≤–∏–Ω–µ–Ω —á–µ–∫–∞—Ç–∏ –ª–∏—à–µ –Ω–∞—Å—Ç—É–ø–Ω–æ–≥–æ –º–æ–º–µ–Ω—Ç—É —á–∞—Å—É. –í –º–æ–º–µ–Ω—Ç $t+1$ –∞–ª–≥–æ—Ä–∏—Ç–º —Ñ–æ—Ä–º—É—î –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è –Ω–∞ –æ—Å–Ω–æ–≤—ñ –æ—Ç—Ä–∏–º–∞–Ω–æ—ó –≤–∏–Ω–∞–≥–æ—Ä–æ–¥–∏ $r_{t+1}$ —Ç–∞ –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è $V(s_{t+1})$. –ù–∞–π–ø—Ä–æ—Å—Ç—ñ—à–∏–π TD-–∞–ª–≥–æ—Ä–∏—Ç–º, –∑–Ω–∞–Ω–∏–π —è–∫ TD(0), –≤–∏–≥–ª—è–¥–∞—î –Ω–∞—Å—Ç—É–ø–Ω–∏–º —á–∏–Ω–æ–º:
\begin{equation}
V(s_t) \leftarrow V(s_t) + \alpha\left[r_{t+1} + \gamma V(s_{t+1}) - V(s_t)\right].
\end{equation}

– –∞–Ω—ñ—à–µ –≤–∂–µ –∑–≥–∞–¥—É–≤–∞–ª–æ—Å—è, —â–æ
\begin{eqnarray}
V^\pi(s) & = & E_\pi \Bigl\{R_t \Big| s_t = s \Bigr\} \label{MC_target}\\
			& = & E_\pi \Biggl\{\sum_{k=0}^{\infty}\gamma^k r_{t+k+1} \Bigg| s_t = s \Biggr\} \nonumber\\
			& = & E_\pi \left\{r_{t+1} + \gamma\sum_{k=0}^{\infty}\gamma^k r_{t+k+2} \Bigg| s_t = s \right\} \nonumber\\
			& = & E_\pi \left\{r_{t+1} + \gamma V^\pi(s_{t+1}) \bigg| s_t = s \right\}.\label{DP_target}
\end{eqnarray}

–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –∑ –ø–æ–ø–µ—Ä–µ–¥–Ω—å–æ–≥–æ –∑–∞–ø–∏—Å—É –≤–∏–¥–Ω–æ, —â–æ MC-–º–µ—Ç–æ–¥ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è –∑ (\ref{MC_target}) —è–∫ –±–∞–∂–∞–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è —Ü—ñ–Ω–Ω—ñ—Å–Ω–æ—ó —Ñ—É–Ω–∫—Ü—ñ—ó, –∞ DP-–º–µ—Ç–æ–¥ "--- –∑ (\ref{DP_target}). –ë–∞–∂–∞–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è –¥–ª—è MC —î –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è–º, –æ—Å–∫—ñ–ª—å–∫–∏ –Ω–µ–≤—ñ–¥–æ–º–µ –º–∞—Ç–µ–º–∞—Ç–∏—á–Ω–µ —Å–ø–æ–¥—ñ–≤–∞–Ω–Ω—è. –í –≤–∏–ø–∞–¥–∫—É –∑ DP –≤–æ–Ω–æ —î –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è–º –Ω–µ —á–µ—Ä–µ–∑ –º–∞—Ç—Å–ø–æ–¥—ñ–≤–∞–Ω–Ω—è, —è–∫–µ –≤—ñ–¥–æ–º–µ –∑ –º–æ–¥–µ–ª—ñ –¥–∏–Ω–∞–º—ñ–∫–∏ —Å–µ—Ä–µ–¥–æ–≤–∏—â–∞, –∞ —á–µ—Ä–µ–∑ —Ç–µ, —â–æ —Å–∞–º–µ –∑–Ω–∞—á–µ–Ω–Ω—è $V^\pi(s_{t+1})$ –Ω–µ–≤—ñ–¥–æ–º–µ. –î–ª—è TD-–º–µ—Ç–æ–¥—É –º–∏ –∂ –æ—Ç—Ä–∏–º—É—î–º–æ –Ω–∞–±–ª–∏–∂–µ–Ω–Ω—è —ñ —á–µ—Ä–µ–∑ –Ω–µ–≤—ñ–¥–æ–º—ñ—Å—Ç—å –º–∞—Ç—Å–ø–æ–¥—ñ–≤–∞–Ω–Ω—è, —ñ —á–µ—Ä–µ–∑ –Ω–µ–≤—ñ–¥–æ–º—ñ—Å—Ç—å —Ä–µ–∞–ª—å–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–Ω—è $V^\pi(s_{t+1})$. –í —Ç–∞–±–ª.~\ref{TD0_algorithm} –ø–æ–¥–∞–Ω–∏–π –∞–ª–≥–æ—Ä–∏—Ç–º TD(0).

\subsubsection{Sarsa: On-Policy TD Control}
